namespace d_array_string_sort;
class Program
{
    static void Main(string[] args)
    {
        string[] names = new string[5] { "Ram", "Sita", "Hari", "Kamal", "Nita" };
        Array.Sort(names);
        Array.Reverse(names);
        for (int i = 0; i < names.Length; i++)
        {
            Console.WriteLine("Name [{0}] = {1}", i, names[i]);
        }
    }
}
